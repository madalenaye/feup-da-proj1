44void MinHeap<K,V>::upHeap(int i) {

51// Make a value go "down the tree" until it reaches its position
52template <class K, class V>
53void MinHeap<K,V>::downHeap(int i) {
54 while (LEFT(i) <= size) { // while within heap limits
55 int j = LEFT(i);
56 if (RIGHT(i)<=size && a[RIGHT(i)].value < a[j].value) j = RIGHT(i); // choose smaller child
